create table if not exists  `database`.`myerror`(
       `T` bool not null default 1,
       `F` bool not null default 0,

delimiter$$
CREATE TRIGGER date_of_transaction  BEFORE INSERT
ON transaction
FOR EACH ROW
BEGIN
   IF (NEW.date_of_transaction > CURDATE()) THEN 
   INSERT INTO myerror(T)

   END IF;
end$$
