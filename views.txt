1η(ΠΩΛΗΣΕΙΣ ΑΝΑ ΚΑΤΑΣΤΗΜΑ ΑΝΑ ΚΑΤΗΓΟΡΙΑ)
CREATE VIEW sales_type_store
AS
SELECT p.transaction_id_purchases, p.card_id_purchases, h.product_id_purchases, p.quantity_purchases, h.type_id_has, h.store_id_has  
FROM purchases AS p, has AS h, product AS q
WHERE p.product_id_purchases = h.product_id_has
      AND h.product_id_has = q.product_id
      ORDER BY h.store_id_has, h.type_id_has




2η(ΠΛΗΡΟΦΟΡΙΕΣ ΓΙΑ ΤΟΝ ΠΕΛΑΤΗ)
CREATE VIEW who_is_the_customer
AS 
SELECT DINSTICT c.card_id, c.first_name, c._second_name, c.last_name, c.marital_status, c.sex, c.phone_number
FROM customer AS c
